  <button onclick="login()">Login</button>
  <hr>
  <button onclick="showStudentRegister()">Create Student Account</button>
  <button onclick="showTeacherRegister()">Create Teacher Account</button>

<div class="container" id="registerPage" style="display:none;">
  <h3 id="registerTitle"></h3>
  <input type="text" id="regUsername" placeholder="Username">
  <input type="password" id="regPassword" placeholder="Password">
  <input type="password" id="teacherCode" placeholder="Enter Teacher Access Code" style="display:none;">
  <button onclick="register()">Register</button>
  <button onclick="goBack()">Back</button>
</div>

<div class="container" id="dashboard" style="display:none;">
  <h2 id="welcome"></h2>
  <div id="teacherPanel" style="display:none;">
    <h3>Teacher Panel</h3>
    <input type="text" id="noticeInput" placeholder="Enter Notice">
    <button onclick="addNotice()">Post Notice</button>

    <input type="text" id="attendanceInput" placeholder="Student Username">
    <input type="number" id="attendanceValue" placeholder="Attendance %">
    <button onclick="addAttendance()">Update Attendance</button>

    <input type="text" id="marksStudent" placeholder="Student Username">
    <input type="number" id="marksValue" placeholder="Marks %">
    <button onclick="addMarks()">Update Marks</button>
  </div>

  <div id="studentPanel">
    <h3>Your Data</h3>
    <p id="studentAttendance"></p>
    <p id="studentMarks"></p>
    <h4>Notices:</h4>
    <div id="notices"></div>
  </div>

  <button onclick="logout()">Logout</button>
</div>

<script>
const TEACHER_SECRET = "9858"; // Hidden teacher access code

let currentUser = "";
let currentRole = "";

function showStudentRegister(){
  document.getElementById("loginPage").style.display="none";
  document.getElementById("registerPage").style.display="block";
  document.getElementById("registerTitle").innerText="Student Registration";
  document.getElementById("teacherCode").style.display="none";
  currentRole="student";
}

function showTeacherRegister(){
  document.getElementById("loginPage").style.display="none";
  document.getElementById("registerPage").style.display="block";
  document.getElementById("registerTitle").innerText="Teacher Registration";
  document.getElementById("teacherCode").style.display="block";
  currentRole="teacher";
}

function goBack(){
  document.getElementById("registerPage").style.display="none";
  document.getElementById("loginPage").style.display="block";
}

function register(){
  let user=document.getElementById("regUsername").value;
  let pass=document.getElementById("regPassword").value;

  if(currentRole=="teacher"){
    let code=document.getElementById("teacherCode").value;
    if(code !== TEACHER_SECRET){
      alert("Invalid Teacher Code");
      return;
    }
  }

  localStorage.setItem(user, JSON.stringify({
    password:pass,
    role:currentRole,
    attendance:0,
    marks:0
  }));

  alert("Account Created!");
  goBack();
}

function login(){
  let user=document.getElementById("username").value;
  let pass=document.getElementById("password").value;
  let data=JSON.parse(localStorage.getItem(user));

  if(data && data.password==pass){
    currentUser=user;
    currentRole=data.role;
    showDashboard();
  } else {
    alert("Invalid Login");
  }
}

function showDashboard(){
  document.getElementById("loginPage").style.display="none";
  document.getElementById("dashboard").style.display="block";
  document.getElementById("welcome").innerText="Welcome "+currentUser;

  if(currentRole=="teacher"){
    document.getElementById("teacherPanel").style.display="block";
    document.getElementById("studentPanel").style.display="none";
  } else {
    document.getElementById("teacherPanel").style.display="none";
    document.getElementById("studentPanel").style.display="block";
    loadStudentData();
  }
}

function addNotice(){
  let notice=document.getElementById("noticeInput").value;
  let allNotices=JSON.parse(localStorage.getItem("notices"))||[];
  allNotices.push(notice);
  localStorage.setItem("notices",JSON.stringify(allNotices));
  alert("Notice Posted");
}

function addAttendance(){
  let student=document.getElementById("attendanceInput").value;
  let value=document.getElementById("attendanceValue").value;
  let data=JSON.parse(localStorage.getItem(student));
  if(data){
    data.attendance=value;
    localStorage.setItem(student,JSON.stringify(data));
    alert("Attendance Updated");
  }
}

function addMarks(){
  let student=document.getElementById("marksStudent").value;
  let value=document.getElementById("marksValue").value;
  let data=JSON.parse(localStorage.getItem(student));
  if(data){
    data.marks=value;
    localStorage.setItem(student,JSON.stringify(data));
    alert("Marks Updated");
  }
}

function loadStudentData(){
  let data=JSON.parse(localStorage.getItem(currentUser));
  document.getElementById("studentAttendance").innerText="Attendance: "+data.attendance+"%";
  document.getElementById("studentMarks").innerText="Marks: "+data.marks+"%";

  let notices=JSON.parse(localStorage.getItem("notices"))||[];
  document.getElementById("notices").innerHTML=notices.join("<br>");
}

function logout(){
  location.reload();
}
</script>
